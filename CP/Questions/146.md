### Trick
- use three things:
	- size
	- array for nodes
	- head and tails

### Code:
```java
class Node {
    int key;
    int val;
    Node back;
    Node front;
}

class LRUCache {
    private int sz;
    private Node head;
    private Node tail;
    private final int max;
    private Node[] arr;

    public LRUCache(int capacity) {
        max = capacity;
        sz = 0;
        head = null;
        arr = new Node[100100];
        Arrays.fill(arr, null);
    }

    private boolean exist(int key){
        return (arr[key]!=null);
    }

    private void move(int key){
        Node temp = arr[key];
        if(temp==head){
            return;
        }

        if(temp==tail){
            tail = tail.front;
            tail.back = null;
        }else {
            Node b = temp.back;
            Node f = temp.front;
            b.front = f;
            f.back = b;
        }

        temp.back = temp.front = null;

        head.front = temp;
        temp.back = head;

        head = head.front;
    }

    private int getVal(int key){
        Node temp = arr[key];
        return temp.val;
    }

    private void update(int key, int value){
        Node temp = arr[key];
        temp.val = value;
    }

    private boolean full(){
        return (sz>=max);
    }

    private boolean isEmpty(){
        return (sz<=0);
    }

    private void evict(){
        if(isEmpty()){
            return;
        }

        if(tail==head){
            sz = 0;
            arr[tail.key] = null;
            tail = head = null;
            return;
        }

        Node temp = tail.front;
        temp.back = null;
        tail.front = null;
        arr[tail.key] = null;
        tail = temp;
        sz--;
    }

    private void putVal(int key, int val){
        if(full()){
            return;
        }

        Node temp = new Node();
        temp.key = key;
        temp.val = val;
        temp.front = temp.back = null;
        if(isEmpty()){
            head = tail = temp;
            sz = 1;
            arr[key] = temp;
            return;
        }
        head.front = temp;
        temp.back = head;
        head = head.front;
        sz++;
        arr[key] = temp;
    }
    
    public int get(int key) {
        if(!exist(key)){
            return -1;
        }
        
        move(key);
        return getVal(key);
    }
    
    public void put(int key, int value) {
        if(max==0){
            return;
        }

        if(exist(key)){
            update(key,value);
            move(key);
        }else{
            if(full()){
                evict();
            }
            putVal(key,value);
        }
    }
}

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache obj = new LRUCache(capacity);
 * int param_1 = obj.get(key);
 * obj.put(key,value);
 */
```
