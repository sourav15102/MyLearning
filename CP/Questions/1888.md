
### Trick:
1. be it "010" pattern or "101" pattern, choose one and calculate incorrect and correct positions.
2. so, minimum(incorrect, correct) will be the answer at any point.

### Code:
```java
class Solution {
    public int minFlips(String s) {
        int l = s.length();
        char[] sc = s.toCharArray();
        int[] judge = new int[2*l-1];
        char one = '1';
        char zero = '0';
        char ch;
        int cor = 0;
        int incor = 0;
        int ans = l;
        int i = 0;
        int j = 0;

        for(;j<l;j++){
            if(j%2==0){
                ch = zero;
            }else{
                ch = one;
            }
            if(sc[j]==ch){
                cor++;
                judge[j]=1;
            }else{
                incor++;
                judge[j]=0;
            }
        }
        ans = Math.min(ans, Math.min(cor,incor));
        l = 2*l-1;
        for(;j<l;j++){
            if(j%2==0){
                ch = zero;
            }else{
                ch = one;
            }
            if(sc[i]==ch){
                cor++;
            }else{
                incor++;
            }
            if(judge[i]==0){
                incor--;
            }else{
                cor--;
            }
            i++;
            ans = Math.min(ans, Math.min(cor,incor));
        }
        return ans;
    }
}
```
