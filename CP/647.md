class Solution {
    private int sol(char[] a, int x, int y){
        if(x<0 || y>=a.length){
            return 0;
        }
        if(a[x]!=a[y]){
            return 0;
        }

        return 1 + sol(a,x-1,y+1);
    }
     
    public int countSubstrings(String s) {
        char[] a = s.toCharArray();
        int l = a.length;

        int ans = 0;
        for(int i=0;i<l;i++){
            ans += sol(a,i,i);
            ans += sol(a,i,i+1);
        }

        return ans;
    }
}
